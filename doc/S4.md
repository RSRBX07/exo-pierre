- `delegate`
  ```ruby
    class Fruit < ApplicationRecord
      belongs_to :orchard
      # pour appeler le jeu d'un Fruit
      # ne fonctionne pas
      # belongs_to :game, through: :orchard
      def game
        orchard.game
      end
      # ou en une ligne
      delegate :game, to: :orchard
    end
  ```
